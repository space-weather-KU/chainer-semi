from chainer import Chain, ChainList
from chainer import links as L
import numpy as np

class ProbChainList(ChainList):
    def __init__(self, n):
        super(ProbChainList, self).__init__(
            L.Linear(3, 3)
        )
        for m in range(n-1):
            self.add_link(L.Linear(3,3))

# x should be ndarray
    def __call__(self, x):
        for k in range(len(self)):
